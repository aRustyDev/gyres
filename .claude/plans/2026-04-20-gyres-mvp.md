# Gyres MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a working LLM agent harness with the core trait system, permission chain (Oso Polar fork), worktree-aware sessions, telemetry, and a functional LLM agent loop with tool execution.

**Architecture:** Seven-crate workspace. `gyres-core` defines domain-agnostic traits (Agent, Gyre, GyreContext). `gyres-polar` embeds a trimmed Oso Polar engine for permission evaluation. `gyres-runtime` handles worktree-aware sessions and async execution. `gyres-tracing` provides telemetry sinks. `gyres-llm` implements the LLM-specific agent loop with tool registry and context management. `gyres-mcp` bridges MCP servers. `gyres` re-exports everything.

**Tech Stack:** Rust 2024 edition, tokio, serde, reqwest, Oso Polar (forked), tracing crate

```rust
// crates/gyres-core/src/permissions.rs
use std::future::Future;
use std::pin::Pin;

use crate::types::AgentId;

/// What kind of action is being requested.
#[derive(Debug, Clone)]
pub enum ActionKind {
    /// Read-only tool invocation (glob, grep, read).
    Read { tool: String },
    /// Write tool invocation (write, edit).
    Write { tool: String },
    /// Execution tool invocation (bash).
    Execute { tool: String, input: String },
}

impl ActionKind {
    /// Returns true if this action modifies state.
    pub fn is_write(&self) -> bool {
        matches!(self, ActionKind::Write { .. } | ActionKind::Execute { .. })
    }

    /// The tool name for this action.
    pub fn tool(&self) -> &str {
        match self {
            ActionKind::Read { tool } => tool,
            ActionKind::Write { tool } => tool,
            ActionKind::Execute { tool, .. } => tool,
        }
    }
}

/// The resource being acted upon.
#[derive(Debug, Clone)]
pub enum Resource {
    /// No specific resource (e.g., bash command).
    None,
    /// A file path.
    File { path: String },
    /// A URL.
    Url { url: String },
}

/// A request to perform an action, evaluated by the permission chain.
#[derive(Debug, Clone)]
pub struct PermissionRequest {
    /// Which agent is making the request.
    pub agent_id: AgentId,
    /// What action is being requested.
    pub action: ActionKind,
    /// What resource is being acted upon.
    pub resource: Resource,
}

impl PermissionRequest {
    pub fn new(agent_id: AgentId, action: ActionKind, resource: Resource) -> Self {
        Self {
            agent_id,
            action,
            resource,
        }
    }
}

/// The result of evaluating a permission request.
#[derive(Debug, Clone, PartialEq, Eq)]
pub enum Verdict {
    /// Action is allowed.
    Allow,
    /// Action is denied with a reason.
    Deny(String),
    /// This gate has no opinion — pass to the next gate in the chain.
    Defer,
}

impl std::fmt::Display for Verdict {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            Verdict::Allow => write!(f, "allow"),
            Verdict::Deny(reason) => write!(f, "deny: {reason}"),
            Verdict::Defer => write!(f, "defer"),
        }
    }
}

/// A single gate in the permission filter chain.
/// Dyn-compatible for use behind `Arc<dyn PermissionGate>` in GyreContext.
pub trait PermissionGate: Send + Sync {
    /// Evaluate a permission request.
    fn evaluate(
        &self,
        request: &PermissionRequest,
    ) -> Pin<Box<dyn Future<Output = Verdict> + Send + '_>>;
}
```

```rust
use std::future::Future;
use std::pin::Pin;
use std::time::SystemTime;

use serde::{Deserialize, Serialize};

use crate::error::GyreError;
use crate::types::WorktreePath;

/// Unique identifier for a session.
#[derive(Debug, Clone, PartialEq, Eq, Hash, Serialize, Deserialize)]
pub struct SessionId(pub String);

/// A single turn in a session (one observation + action cycle).
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Turn {
    /// When this turn occurred.
    pub timestamp: SystemTime,
    /// Serialized observation (domain-specific).
    pub observation: serde_json::Value,
    /// Serialized action (domain-specific).
    pub action: serde_json::Value,
    /// Optional serialized feedback.
    pub feedback: Option<serde_json::Value>,
}

/// Metadata about a session (without full state).
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct SessionMeta {
    pub id: SessionId,
    pub created_at: SystemTime,
    pub last_active: SystemTime,
    pub turn_count: usize,
    pub worktree: Option<String>,
}

/// Serializable session state.
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct SessionState {
    pub id: SessionId,
    pub created_at: SystemTime,
    pub last_active: SystemTime,
    pub turns: Vec<Turn>,
```

```rust
// crates/gyres-polar/src/deny_rule.rs
use gyres_core::permissions::{ActionKind, PermissionRequest};

/// A hard-coded deny rule. Checked first, cannot be overridden.
#[derive(Debug, Clone)]
pub struct DenyRule {
    /// Tool name to match.
    pub tool: String,
    /// Glob pattern to match against the input/path.
    pub pattern: String,
}

impl DenyRule {
    pub fn glob(tool: impl Into<String>, pattern: impl Into<String>) -> Self {
        Self {
            tool: tool.into(),
            pattern: pattern.into(),
        }
    }

    pub fn matches(&self, request: &PermissionRequest) -> bool {
        let tool_matches = request.action.tool() == self.tool;
        if !tool_matches {
            return false;
        }

        let input = match &request.action {
            ActionKind::Execute { input, .. } => input.as_str(),
            ActionKind::Read { .. } | ActionKind::Write { .. } => {
                match &request.resource {
                    gyres_core::permissions::Resource::File { path } => path.as_str(),
                    _ => return false,
                }
            }
        };

        glob_match(&self.pattern, input)
    }
}

/// Simple glob matching (supports `*` as wildcard).
fn glob_match(pattern: &str, input: &str) -> bool {
    let parts: Vec<&str> = pattern.split('*').collect();
    if parts.len() == 1 {
        return pattern == input;
    }

    let mut pos = 0;
    for (i, part) in parts.iter().enumerate() {
        if part.is_empty() {
            continue;
        }
        match input[pos..].find(part) {
            Some(found) => {
                if i == 0 && found != 0 {
                    return false; // Must match from start if no leading *
                }
                pos += found + part.len();
            }
            None => return false,
        }
    }

    // If pattern doesn't end with *, input must end exactly
    if !pattern.ends_with('*') {
        return pos == input.len();
    }

    true
}
```

```rust
// crates/gyres-polar/src/approval_cache.rs
use std::collections::HashMap;
use std::sync::RwLock;
use std::time::{Instant, SystemTime};

use gyres_core::permissions::{PermissionRequest, Verdict};

/// Scope of an approval entry.
#[derive(Debug, Clone, PartialEq, Eq)]
pub enum ApprovalScope {
    /// Valid only for the current session.
    Session,
    /// Persisted across sessions.
    Persistent,
}

/// A cached approval decision.
#[derive(Debug, Clone)]
pub struct ApprovalEntry {
    /// Glob pattern for the tool name.
    pub tool_pattern: String,
    /// Glob pattern for the input/path.
    pub input_pattern: String,
    pub scope: ApprovalScope,
    pub verdict: Verdict,
    pub granted_at: SystemTime,
}

/// Caches HITL approval decisions. Session-scoped by default,
/// entries can be promoted to persistent.
pub struct ApprovalCache {
    entries: RwLock<Vec<ApprovalEntry>>,
}

impl ApprovalCache {
    pub fn new() -> Self {
        Self {
            entries: RwLock::new(Vec::new()),
        }
    }

    /// Check if a request matches a cached approval.
    pub fn check(&self, request: &PermissionRequest) -> Option<Verdict> {
        let entries = self.entries.read().ok()?;
        for entry in entries.iter() {
            if entry.tool_pattern == request.action.tool() {
                return Some(entry.verdict.clone());
            }
        }
        None
    }

    /// Add an approval entry.
    pub fn approve(&self, entry: ApprovalEntry) {
        if let Ok(mut entries) = self.entries.write() {
            entries.push(entry);
        }
    }

    /// Promote a session-scoped entry to persistent.
    pub fn promote(&self, tool_pattern: &str) -> bool {
        if let Ok(mut entries) = self.entries.write() {
            for entry in entries.iter_mut() {
                if entry.tool_pattern == tool_pattern
                    && entry.scope == ApprovalScope::Session
                {
                    entry.scope = ApprovalScope::Persistent;
                    return true;
                }
            }
        }
        false
    }

    /// Get all persistent entries (for serialization).
    pub fn persistent_entries(&self) -> Vec<ApprovalEntry> {
        self.entries
            .read()
            .map(|entries| {
                entries
                    .iter()
                    .filter(|e| e.scope == ApprovalScope::Persistent)
                    .cloned()
                    .collect()
            })
            .unwrap_or_default()
    }
}
```

```rust
// crates/gyres-runtime/src/session/json_store.rs
use std::future::Future;
use std::path::PathBuf;
use std::pin::Pin;

use gyres_core::error::GyreError;
use gyres_core::state::{SessionId, SessionMeta, SessionState, StateStore};

/// File-based session store using JSON files.
/// One file per session: `{base_dir}/{session_id}.json`
pub struct JsonFileStore {
    base_dir: PathBuf,
}

impl JsonFileStore {
    pub fn new(base_dir: PathBuf) -> Self {
        Self { base_dir }
    }

    fn session_path(&self, id: &SessionId) -> PathBuf {
        self.base_dir.join(format!("{}.json", id.0))
    }
}

impl StateStore for JsonFileStore {
    fn save_session(
        &self,
        id: &SessionId,
        state: &SessionState,
    ) -> Pin<Box<dyn Future<Output = Result<(), GyreError>> + Send + '_>> {
        let path = self.session_path(id);
        let state = state.clone();
        Box::pin(async move {
            if let Some(parent) = path.parent() {
                tokio::fs::create_dir_all(parent)
                    .await
                    .map_err(|e| GyreError::State(e.to_string()))?;
            }
            let json = serde_json::to_string_pretty(&state)
                .map_err(|e| GyreError::State(e.to_string()))?;
            tokio::fs::write(&path, json)
                .await
                .map_err(|e| GyreError::State(e.to_string()))?;
            Ok(())
        })
    }

    fn load_session(
        &self,
        id: &SessionId,
    ) -> Pin<Box<dyn Future<Output = Result<Option<SessionState>, GyreError>> + Send + '_>> {
        let path = self.session_path(id);
        Box::pin(async move {
            if !path.exists() {
                return Ok(None);
            }
            let json = tokio::fs::read_to_string(&path)
                .await
                .map_err(|e| GyreError::State(e.to_string()))?;
            let state: SessionState = serde_json::from_str(&json)
                .map_err(|e| GyreError::State(e.to_string()))?;
            Ok(Some(state))
        })
    }

    fn list_sessions(
        &self,
    ) -> Pin<Box<dyn Future<Output = Result<Vec<SessionMeta>, GyreError>> + Send + '_>> {
        let base_dir = self.base_dir.clone();
        Box::pin(async move {
            let mut sessions = Vec::new();
            let mut entries = tokio::fs::read_dir(&base_dir)
                .await
                .map_err(|e| GyreError::State(e.to_string()))?;

            while let Some(entry) = entries.next_entry().await
                .map_err(|e| GyreError::State(e.to_string()))? {
                let path = entry.path();
                if path.extension().is_some_and(|ext| ext == "json") {
                    let json = tokio::fs::read_to_string(&path)
                        .await
                        .map_err(|e| GyreError::State(e.to_string()))?;
                    if let Ok(state) = serde_json::from_str::<SessionState>(&json) {
                        sessions.push(state.meta());
                    }
                }
            }
            Ok(sessions)
        })
    }
}
```

```rust
            .await
            .map_err(|e| GyreError::State(format!("git not found: {e}")))?;

        if !output.status.success() {
            return Ok(None);
        }

        let worktree_root = String::from_utf8_lossy(&output.stdout).trim().to_string();

        let branch = run_git(path, &["rev-parse", "--abbrev-ref", "HEAD"]).await?;
        let commit_hash = run_git(path, &["rev-parse", "HEAD"]).await?;
        let is_main = run_git(path, &["rev-parse", "--git-common-dir"]).await
            .map(|common| {
                let git_dir = run_git_sync(path, &["rev-parse", "--git-dir"]);
                // Main worktree: .git dir == common dir
                common.trim_end_matches('/') == git_dir.trim_end_matches('/')
            })
            .unwrap_or(true);

        let remote_url = run_git(path, &["remote", "get-url", "origin"]).await.ok();

        Ok(Some(GitContext {
            branch,
            commit_hash,
            worktree_root: std::path::PathBuf::from(worktree_root),
            is_main_worktree: is_main,
            remote_url,
        }))
    }
}

async fn run_git(path: &std::path::Path, args: &[&str]) -> Result<String, GyreError> {
    let output = tokio::process::Command::new("git")
        .args(args)
        .current_dir(path)
        .output()
        .await
        .map_err(|e| GyreError::State(format!("git error: {e}")))?;

    if !output.status.success() {
        return Err(GyreError::State(format!(
            "git {} failed: {}",
            args.join(" "),
            String::from_utf8_lossy(&output.stderr)
        )));
    }

    Ok(String::from_utf8_lossy(&output.stdout).trim().to_string())
}

fn run_git_sync(path: &std::path::Path, args: &[&str]) -> String {
    std::process::Command::new("git")
        .args(args)
        .current_dir(path)
        .output()
        .map(|o| String::from_utf8_lossy(&o.stdout).trim().to_string())
        .unwrap_or_default()
}
```

```rust
/// Handler function type for tool execution.
pub type ToolHandler = Box<
    dyn Fn(&serde_json::Value) -> Pin<Box<dyn Future<Output = Result<String, GyreError>> + Send>>
        + Send
        + Sync,
>;

/// Definition of a tool that an agent can use.
pub struct ToolDef {
    pub name: String,
    pub description: String,
    pub parameters: serde_json::Value,
    pub is_read_only: bool,
    pub handler: ToolHandler,
}

/// Registry of available tools.
pub struct ToolRegistry {
    tools: HashMap<String, ToolDef>,
}

impl ToolRegistry {
    pub fn new() -> Self {
        Self {
            tools: HashMap::new(),
        }
    }

    pub fn register(&mut self, tool: ToolDef) {
        self.tools.insert(tool.name.clone(), tool);
    }

    pub fn get(&self, name: &str) -> Option<&ToolDef> {
        self.tools.get(name)
    }

    /// Get JSON schemas for all registered tools (for sending to LLM).
    pub fn schemas(&self) -> Vec<ToolSchema> {
        self.tools
            .values()
            .map(|t| ToolSchema {
                name: t.name.clone(),
                description: t.description.clone(),
                parameters: t.parameters.clone(),
            })
            .collect()
    }

    /// Execute a tool by name.
    pub async fn execute(
        &self,
        name: &str,
        params: &serde_json::Value,
    ) -> Result<String, GyreError> {
        let tool = self.tools.get(name).ok_or_else(|| {
            GyreError::Agent(format!("unknown tool: {name}"))
        })?;
        (tool.handler)(params).await
    }
}

impl Default for ToolRegistry {
    fn default() -> Self {
        Self::new()
    }
}
```

```rust
            provider,
            registry,
            history: Vec::new(),
            stop: false,
        }

    pub fn set_input(&mut self, message: Message) {
        self.history.push(message);
    }
}

impl Agent for LlmAgent {
    type Observation = LlmObservation;
    type Action = LlmAction;
    type Feedback = LlmFeedback;
    type Error = GyreError;

    fn step(
        &mut self,
        obs: &LlmObservation,
    ) -> impl std::future::Future<Output = Result<LlmAction, GyreError>> + Send {
        self.history.push(obs.clone());
        let messages = self.history.clone();
        let schemas = self.registry.schemas();
        let provider = self.provider.clone();

        async move {
            let response = provider.complete(&messages, &schemas).await?;
            Ok(response)
        }
    }

    fn feedback(&mut self, fb: &LlmFeedback) {
        self.history.push(fb.clone());
    }

        self.stop
    }
```

```rust
// crates/gyres-llm/src/gyre.rs
use gyres_core::context::GyreContext;
use gyres_core::error::GyreError;
use gyres_core::gyre::Gyre;
use gyres_core::agent::Agent;

use crate::agent::LlmAgent;
use crate::message::{Message, Role};

/// The outcome of a conversation run.
#[derive(Debug, Clone)]
pub struct ConversationOutcome {
    /// All messages in the conversation.
    pub messages: Vec<Message>,
    /// Number of turns (assistant responses) in this run.
    pub turn_count: usize,
}

/// Drives an LlmAgent through a conversation loop.
/// Handles tool call dispatch and permission checking.
pub struct ConversationGyre {
    max_turns: usize,
}

impl ConversationGyre {
    /// Create a gyre that runs exactly one turn (no tool loops).
    pub fn single_turn() -> Self {
        Self { max_turns: 1 }
    }

    /// Create a gyre with a maximum number of agent turns.
    pub fn with_max_turns(max_turns: usize) -> Self {
        Self { max_turns }
    }
}

impl Gyre<LlmAgent> for ConversationGyre {
    type Outcome = ConversationOutcome;

    fn run(
        &mut self,
        agent: &mut LlmAgent,
        ctx: &mut GyreContext,
    ) -> impl std::future::Future<Output = Result<ConversationOutcome, GyreError>> + Send {
        async move {
            let mut turn_count = 0;

            // The initial observation is already in agent.history via set_input()
            // We need a dummy observation to kick off the first step
            let initial = agent.history.last().cloned().unwrap_or_else(|| {
                Message::user("")
            });

            let mut last_obs = initial;

            loop {
                if turn_count >= self.max_turns || agent.should_stop() {
                    break;
                }

                let response = agent.step(&last_obs).await?;
                agent.history.push(response.clone());
                turn_count += 1;

                // If the response has tool calls, execute them
                if response.tool_calls.is_empty() {
                    break; // No tool calls = conversation turn complete
                }

                // Execute each tool call
                for call in &response.tool_calls {
                    let result = agent.registry.execute(&call.name, &call.arguments).await;
                    let tool_msg = Message::tool_result(crate::message::ToolResult {
                        call_id: call.id.clone(),
                        output: result.unwrap_or_else(|e| format!("Error: {e}")),
                        is_error: result.is_err(),
                    });
                    agent.feedback(&tool_msg);
                    last_obs = tool_msg;
                }
            }

            Ok(ConversationOutcome {
                messages: agent.history.clone(),
                turn_count,
            })
        }
    }
}
```

```rust
//! End-to-end integration test: wire up all crates into a working agent.

use gyres_core::config::Config;

/// Mock provider that calls the "echo" tool then returns the result.
struct ToolCallingProvider {
    call_count: std::sync::atomic::AtomicUsize,
}

#[async_trait::async_trait]
impl Provider for ToolCallingProvider {
    async fn complete(
        &self,
        messages: &[Message],
        _tools: &[ToolSchema],
    ) -> Result<Message, GyreError> {
        let count = self.call_count.fetch_add(1, std::sync::atomic::Ordering::SeqCst);

        if count == 0 {
            // First call: invoke the echo tool
            Ok(Message::assistant_with_tool_calls(
                "Let me echo that.",
                vec![ToolCall {
                    id: "call_1".into(),
                    name: "echo".into(),
                    arguments: serde_json::json!({"text": "integration test works"}),
                }],
            ))
        } else {
            // Second call: return final response
            let last = messages.last().unwrap();
            Ok(Message::assistant(format!(
                "Done. Tool said: {}",
                match &last.content {
                    gyres_llm::message::Content::Text(t) => t.as_str(),
                    _ => "unknown",
                }
            )))
        }
    }
}

struct NullStore;
impl StateStore for NullStore {
    fn save_session(&self, _: &SessionId, _: &SessionState) -> Pin<Box<dyn Future<Output = Result<(), GyreError>> + Send + '_>> {
        Box::pin(async { Ok(()) })
    }
    fn load_session(&self, _: &SessionId) -> Pin<Box<dyn Future<Output = Result<Option<SessionState>, GyreError>> + Send + '_>> {
        Box::pin(async { Ok(None) })
    }
    fn list_sessions(&self) -> Pin<Box<dyn Future<Output = Result<Vec<SessionMeta>, GyreError>> + Send + '_>> {
        Box::pin(async { Ok(vec![]) })
    }
}

fn echo_tool() -> ToolDef {
    ToolDef {
        name: "echo".into(),
        description: "Echoes text back".into(),
        parameters: serde_json::json!({
            "type": "object",
            "properties": { "text": { "type": "string" } },
            "required": ["text"]
        }),
        is_read_only: true,
        handler: Box::new(|params| {
            Box::pin(async move {
                Ok(params["text"].as_str().unwrap_or("").to_string())
            })
        }),
    }
}

#[tokio::test]
async fn test_full_agent_loop_with_tool_call() {
    // Wire up: permission chain + null store + noop telemetry
    let permissions = PermissionChain::builder()
        .deny_rule(DenyRule::glob("bash", "rm -rf *"))
        .default_allow()
        .build();

    let mut registry = ToolRegistry::new();
    registry.register(echo_tool());

    let provider = Arc::new(ToolCallingProvider {
        call_count: std::sync::atomic::AtomicUsize::new(0),
    });

    let mut agent = LlmAgent::new(provider, registry);
    agent.set_input(Message::user("Please echo 'integration test works'"));

    let mut gyre = ConversationGyre::with_max_turns(5);

    let mut ctx = GyreContext {
        permissions: Arc::new(permissions),
        state: Arc::new(NullStore),
        config: Arc::new(Config::default()),
        telemetry: Arc::new(NoopTelemetry),
    };

    let outcome = gyre.run(&mut agent, &mut ctx).await.unwrap();

    // Should have: user msg, assistant w/ tool call, tool result, final assistant
    assert!(outcome.turn_count >= 2);
    let last = outcome.messages.last().unwrap();
    assert_eq!(last.role, Role::Assistant);
    assert!(matches!(&last.content,
        gyres_llm::message::Content::Text(t) if t.contains("integration test works")
    ));
}
```
